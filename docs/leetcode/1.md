# 1. 两数之和

```ts
function twoSum(nums: number[], target: number): number[] {
  const map = new Map<number, number>();
  for (const [index, num] of nums.entries()) {
    const other = target - num;
    if (map.has(other)) {
      return [map.get(other), index];
    }
    map.set(num, index);
  }
  return [0, 1];
}
```
