UseEntry #

ts
type UseEntry = UseEntryObject | ((info: UseEntryInfo) => UseEntryObject);

interface UseEntryObject {
  loader: string;
  // loader 配置，会被传给 loader
  options?: string | object;
  // options 的别名，已废弃
  query?: string | object;
}

interface UseEntryInfo {
  compiler?: string;
  issuer: string;
  realResource: string;
  resource: string;
}
